#CREATE DATABASE

CREATE DATABASE bankdb;
USE bankdb;

#CREATE TABLES:
#CUSTOMER TABLE

CREATE TABLE CUSTOMERS(
CUSTOMER_ID VARCHAR(10) PRIMARY KEY,
CUSTOMER_NAME VARCHAR(30),
CUSTOMER_PHONE VARCHAR(10),
CUSTOMER_EMAIL VARCHAR(40),
CUSTOMER_CITY VARCHAR(50),
CUSTOMER_USERNAME VARCHAR(30),
CUSTOMER_PASSWORD VARCHAR(20)
);

#ACCOUNTS TABLE

CREATE TABLE ACCOUNTS(
ACCOUNT_NUMBER VARCHAR(10) PRIMARY KEY,
CUSTOMER_ID VARCHAR(10),
ACCOUNT_TYPE VARCHAR(15),
ACCOUNT_BALANCE VARCHAR(20),
UPI_ID VARCHAR(25),
ACCOUNT_STATUS VARCHAR(30),
FOREIGN KEY(CUSTOMER_ID) REFERENCES CUSTOMERS(CUSTOMER_ID)
);

#TRANSACTIONS TABLE

CREATE TABLE TRANSACTIONS(
TRANSACTION_ID VARCHAR(10) PRIMARY KEY,
SENDER_ACCOUNT_NUMBER VARCHAR(10),
RECEIVER_ACCOUNT_NUMBER VARCHAR(10),
AMOUNT VARCHAR(20),
TXN_TIMESTAMP TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
TXN_STATUS VARCHAR(30)
);
